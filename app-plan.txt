  <div class="container">
    <h1>JobSave</h1>

    <section>
      <h2>Personal Information</h2>

      <label>Address</label>

      <input type="email" id="email" placeholder="jane@email.com">

      <button onclick="savePersonal()">Save Personal Info</button>

    <!-- Work History -->
    <section>
      <h2>Work History & Resume Details</h2>
      <div id="work-list"></div>
      
      <label>Job Title</label>

      <label>Company</label>
      <input type="text" id="company" placeholder="Tech Corp">

      <label>Description / Achievements</label>
      <textarea id="job-desc" rows="3" placeholder="Built features, improved performance..."></textarea>

      <button onclick="addWorkEntry()">Add Work Entry</button>
    </section>

    <!-- Saved Answers -->
    <section>
      <h2>Saved Answers for Common Questions</h2>
      <div id="answers-list"></div>
      
      <label>New Answer</label>
      <textarea id="new-answer" rows="4" placeholder="Type your answer here (e.g. Tell me about yourself...)"></textarea>

      <button onclick="addAnswer()">Add Saved Answer</button>
    </section>

    <div class="status" id="status"></div>
  </div>

  <script>
    // Auto-save and load all data
    function loadAllData() {
      // Personal info
      document.getElementById('name').value = localStorage.getItem('name') || '';
      document.getElementById('address').value = localStorage.getItem('address') || '';
      document.getElementById('phone').value = localStorage.getItem('phone') || '';
      document.getElementById('email').value = localStorage.getItem('email') || '';

      // Work history
      const workList = document.getElementById('work-list');
      workList.innerHTML = '';
      const workData = JSON.parse(localStorage.getItem('workHistory') || '[]');
      workData.forEach((entry, index) => {
        const div = document.createElement('div');
        div.className = 'item';
        div.innerHTML = `
          <strong>${entry.title} at ${entry.company}</strong><br>
          ${entry.description}
          <button class="delete-btn" onclick="deleteWork(${index})">Delete</button>
        `;
        workList.appendChild(div);
      });

      // Saved answers
      const answersList = document.getElementById('answers-list');
      answersList.innerHTML = '';
      const answersData = JSON.parse(localStorage.getItem('savedAnswers') || '[]');
      answersData.forEach((answer, index) => {
        const div = document.createElement('div');
        div.className = 'item';
        div.innerHTML = `
          ${answer}
          <button class="delete-btn" onclick="deleteAnswer(${index})">Delete</button>
        `;
        answersList.appendChild(div);
      });
    }

    function savePersonal() {
      localStorage.setItem('name', document.getElementById('name').value);
      localStorage.setItem('address', document.getElementById('address').value);
      localStorage.setItem('phone', document.getElementById('phone').value);
      localStorage.setItem('email', document.getElementById('email').value);
      showStatus('Personal information saved!');
    }

    function addWorkEntry() {
      const title = document.getElementById('job-title').value.trim();
      const company = document.getElementById('company').value.trim();
      const desc = document.getElementById('job-desc').value.trim();
      
      if (!title || !company) {
        showStatus('Please enter job title and company');
        return;
      }

      let workHistory = JSON.parse(localStorage.getItem('workHistory') || '[]');
      workHistory.push({ title, company, description: desc });
      localStorage.setItem('workHistory', JSON.stringify(workHistory));

      document.getElementById('job-title').value = '';
      document.getElementById('company').value = '';
      document.getElementById('job-desc').value = '';

      loadAllData();
      showStatus('Work entry added!');
    }

    function deleteWork(index) {
      let workHistory = JSON.parse(localStorage.getItem('workHistory') || '[]');
      workHistory.splice(index, 1);
      localStorage.setItem('workHistory', JSON.stringify(workHistory));
      loadAllData();
      showStatus('Work entry removed');
    }

    function addAnswer() {
      const answerText = document.getElementById('new-answer').value.trim();
      if (!answerText) return;

      let savedAnswers = JSON.parse(localStorage.getItem('savedAnswers') || '[]');
      savedAnswers.push(answerText);
      localStorage.setItem('savedAnswers', JSON.stringify(savedAnswers));

      document.getElementById('new-answer').value = '';
      loadAllData();
      showStatus('Answer saved!');
    }

    function deleteAnswer(index) {
      let savedAnswers = JSON.parse(localStorage.getItem('savedAnswers') || '[]');
      savedAnswers.splice(index, 1);
      localStorage.setItem('savedAnswers', JSON.stringify(savedAnswers));
      loadAllData();
      showStatus('Answer removed');
    }

    function showStatus(message) {
      const statusEl = document.getElementById('status');
      statusEl.textContent = message;
      setTimeout(() => {
        statusEl.textContent = '';
      }, 2500);
    }

    // Load data when page opens + auto-save on input changes for personal info
    window.onload = () => {
      loadAllData();

      // Auto-save personal info as you type
      const personalInputs = ['name', 'address', 'phone', 'email'];
      personalInputs.forEach(id => {
        const el = document.getElementById(id);
        if (el) {
          el.addEventListener('input', () => {
            localStorage.setItem(id, el.value);
          });
        }
      });
    };
  </script>
